def self.count_occurences(key)
  re = /<("[^"]*"|'[^']*'|[^'">])*>/
  result = Hash.new(0)
  @tmp = ''
  Document.all.find_each do |document|
    document.content.split.each do |word|
      word = word.gsub(re, '')
      if word.downcase == key.downcase
        result[word] += 1
        @tmp = word
      end
    end
  end
  result[@tmp]
end
